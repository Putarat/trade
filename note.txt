Project trade_portfolio_scanner {
  database_type: "PostgreSQL"
  Note: '''
  Portfolio scanner with per-scan payment model.
  '''
}

Enum user_status {
  active
  suspended
  deleted
}

Enum provider_type {
  email
  google
  apple
  line
}

Enum plan_type {
  pay_per_use
  package
  subscription
}

Enum payment_status {
  pending
  paid
  failed
  refunded
  canceled
}

Enum payment_method {
  promptpay
  card
  transfer
  wallet
}

Enum currency_code {
  THB
  USD
}

Enum scan_status {
  queued
  processing
  completed
  failed
}

Table users {
  id uuid [pk]
  email varchar(255) [not null, unique]
  password_hash varchar(255) [not null]
  display_name varchar(120)
  status user_status [not null, default: 'active']
  created_at timestamptz [not null, default: `now()`]
  updated_at timestamptz [not null, default: `now()`]

  Note: 'Main user account table'
}

Table user_auth_providers {
  id uuid [pk]
  user_id uuid [not null, ref: > users.id]
  provider provider_type [not null]
  provider_user_id varchar(255) [not null]
  created_at timestamptz [not null, default: `now()`]

  Indexes {
    (provider, provider_user_id) [unique]
    user_id
  }
}

Table user_sessions {
  id uuid [pk]
  user_id uuid [not null, ref: > users.id]
  refresh_token_hash varchar(255) [not null]
  user_agent text
  ip_address varchar(64)
  expires_at timestamptz [not null]
  created_at timestamptz [not null, default: `now()`]

  Indexes {
    user_id
    expires_at
  }
}

Table pricing_plans {
  id uuid [pk]
  name varchar(120) [not null]
  plan_type plan_type [not null]
  price_thb numeric(10,2) [not null]
  scans_included int [not null, default: 0]
  duration_days int
  is_active boolean [not null, default: true]
  created_at timestamptz [not null, default: `now()`]

  Note: 'Examples: Pay-per-scan 20 THB, package 10 scans'
}

Table wallets {
  id uuid [pk]
  user_id uuid [not null, unique, ref: > users.id]
  scan_credits int [not null, default: 0]
  updated_at timestamptz [not null, default: `now()`]
}

Table payments {
  id uuid [pk]
  user_id uuid [not null, ref: > users.id]
  pricing_plan_id uuid [ref: > pricing_plans.id]
  amount_thb numeric(10,2) [not null]
  payment_method payment_method [not null]
  status payment_status [not null, default: 'pending']
  provider_txn_id varchar(255)
  paid_at timestamptz
  created_at timestamptz [not null, default: `now()`]

  Indexes {
    user_id
    status
    created_at
    provider_txn_id [unique]
  }
}

Table scans {
  id uuid [pk]
  user_id uuid [not null, ref: > users.id]
  source_image_url text
  status scan_status [not null, default: 'queued']
  ocr_raw_text text
  error_message text
  score numeric(4,1)
  portfolio_type varchar(30)
  total_value_thb numeric(14,2)
  created_at timestamptz [not null, default: `now()`]
  completed_at timestamptz

  Note: 'One scan job per uploaded screenshot'
  Indexes {
    user_id
    status
    created_at
  }
}

Table scan_charge_logs {
  id uuid [pk]
  scan_id uuid [not null, unique, ref: > scans.id]
  user_id uuid [not null, ref: > users.id]
  charge_model varchar(30) [not null, default: 'pay_per_scan']
  amount_thb numeric(10,2) [not null, default: 20.00]
  payment_id uuid [ref: > payments.id]
  credits_used int [not null, default: 0]
  charged_at timestamptz [not null, default: `now()`]

  Indexes {
    user_id
    charged_at
  }
}

Table user_stocks {
  id uuid [pk]
  user_id uuid [not null, ref: > users.id]
  symbol varchar(20) [not null]
  display_name varchar(120)
  logo_url text
  is_active boolean [not null, default: true]
  created_at timestamptz [not null, default: `now()`]

  Indexes {
    (user_id, symbol) [unique]
    user_id
  }
}

Table scan_items {
  id uuid [pk]
  scan_id uuid [not null, ref: > scans.id]
  user_stock_id uuid [not null, ref: > user_stocks.id]
  quantity numeric(18,6) [not null, default: 0]
  price numeric(18,6) [not null, default: 0]
  currency currency_code [not null, default: 'THB']
  fx_rate_to_thb numeric(18,6) [not null, default: 1]
  value_thb numeric(18,2) [not null, default: 0]
  source varchar(30) [not null, default: 'ocr']
  confidence numeric(5,4)
  created_at timestamptz [not null, default: `now()`]

  Indexes {
    scan_id
    user_stock_id
    (scan_id, user_stock_id)
  }
}

Table stock_name_edits {
  id uuid [pk]
  user_id uuid [not null, ref: > users.id]
  user_stock_id uuid [not null, ref: > user_stocks.id]
  old_symbol varchar(20) [not null]
  new_symbol varchar(20) [not null]
  edited_at timestamptz [not null, default: `now()`]

  Indexes {
    user_id
    user_stock_id
    edited_at
  }
}
